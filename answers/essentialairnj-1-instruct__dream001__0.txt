#include <iostream>
#include <vector>
#include <queue>
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

const int N = 1e5 + 5;

vector<int> g[N];
int dis[N], dis2[N], n, m;

void bfs(int s, int* dist) {
    for (int i = 1; i <= n; i++) dist[i] = -1;
    queue<int> q; q.push(s);
    dist[s] = 0;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v : g[u]) {
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
}

int main() {
    scanf("%d%d", &n, &m);
    if (m < n - 1) { printf("IMPOSSIBLE\n"); return 0; }
    for (int i = 1, u, v; i < n; i++) {
        scanf("%d%d", &u, &v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
    bfs(1, dis);
    int max_node = 1, max_dist = 0;
    for (int i = 1; i <= n; i++)
        if (dis[i] > max_dist) {
            max_node = i;
            max_dist = dis[i];
        }
    bfs(max_node, dis2);
    int ans = 0, cnt = 0;
    for (int i = 1; i <= n; i++)
        if (dis2[i] == max_dist) cnt++;
    for (int i = 1, a, b; i <= m; i++) {
        scanf("%d%d", &a, &b);
        if (dis2[a] + dis2[b] == max_dist) ans++;
    }
    if (cnt == 1) ans = ans / 2 + (ans % 2);
    else ans = (ans + 1) / 2;
    printf("%d\n", ans);
    return 0;
}
